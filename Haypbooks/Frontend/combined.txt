import { render, screen, fireEvent } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { act } from 'react-dom/test-utils'
import PracticeHubPage from '@/app/practice-hub/page'
import PracticeHubLayout from '@/app/practice-hub/layout'
import HomeTopBar from '@/components/practice-hub/HomeTopBar'
import DashboardPage from '@/app/practice-hub/dashboard/page'
import GettingStartedPage from '@/app/practice-hub/dashboard/getting-started/page'
import AnnouncementPage from '@/app/practice-hub/dashboard/announcement/page'

let mockPath = '/practice-hub/dashboard'
jest.mock('next/navigation', () => ({
  usePathname: () => mockPath,
  useRouter: jest.fn(() => ({ push: jest.fn(), replace: jest.fn(), prefetch: jest.fn() }))
}))

describe('Practice Hub Page', () => {
  })

  it('renders practice hub primary header variant and stats (no owner chrome)', () => {
    render(<PracticeHubLayout>{<PracticeHubPage />}</PracticeHubLayout>)

    // PracticeHeader should render brand and controls but *no* company switcher
    expect(screen.getByText('HaypBooks')).toBeInTheDocument()
    expect(screen.getByText('Practice Hub')).toBeInTheDocument()
    // primary bar should be wrapped by the standard page-width container
    const primaryWrapper = screen.getByText('HaypBooks').closest('div.max-w-[var(--ph-main-width)]')
    expect(primaryWrapper).toBeInTheDocument()
    // container should look like a pill (white, rounded, shadow)
    expect(primaryWrapper).toHaveClass('bg-white')
    expect(primaryWrapper).toHaveClass('rounded-2xl')
    expect(primaryWrapper).toHaveClass('shadow')
    // bar element itself should be using the smaller height variable
    const outerBar = primaryWrapper?.closest('div.topbar-floating')
    expect(outerBar).toBeInTheDocument()
    if (outerBar) {
      expect(outerBar.getAttribute('style')).toContain('--hb-topbar-height-sm')
    }
    expect(screen.getByPlaceholderText(/Search clients, work, messages/i)).toBeInTheDocument()
    expect(screen.getByRole('button', { name: /\+ Invite/i })).toBeInTheDocument()
    expect(screen.getByRole('button', { name: /New Work/i })).toBeInTheDocument()
    expect(screen.queryByText('Companies')).toBeNull()
    expect(screen.queryByText('Switch hub')).toBeNull()
    // global Sidebar must not appear
    expect(screen.queryByRole('navigation')).toBeNull()

    // secondary header should show current page (home maps to Home)
    expect(screen.getByText('Home')).toBeInTheDocument()

    // Stats
    expect(screen.getByText('Active Clients')).toBeInTheDocument()
    expect(screen.getByText('28')).toBeInTheDocument()
    expect(screen.getByText('Work Due This Week')).toBeInTheDocument()
    expect(screen.getByText('14')).toBeInTheDocument()
    expect(screen.getByText('7')).toBeInTheDocument()

    // adjuster toggle should exist (edges control)
    expect(screen.getByLabelText('Adjust topbar edges')).toBeInTheDocument()
    // open the panel and check all control labels
    fireEvent.click(screen.getByLabelText('Adjust topbar edges'))
    expect(screen.getByLabelText('Vertical offset')).toBeInTheDocument()
    expect(screen.getByLabelText('Sidebar top gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Sidebar bottom gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Sidebar left gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Home bar vertical offset')).toBeInTheDocument()
    expect(screen.getByLabelText('Home bar horizontal offset')).toBeInTheDocument()
    expect(screen.getByLabelText('Home bar total width')).toBeInTheDocument()
    expect(screen.getByLabelText('Primary bar left screen gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Primary bar right screen gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Secondary bar left screen gap')).toBeInTheDocument()
    expect(screen.getByLabelText('Secondary bar right screen gap')).toBeInTheDocument()

    // changing home bar sliders should update CSS variables
    const vertCtl = screen.getByLabelText('Home bar vertical offset') as HTMLInputElement
    fireEvent.change(vertCtl, { target: { value: '18' } })
    expect(document.documentElement.style.getPropertyValue('--ph-home-topbar-vert')).toBe('18px')
    const horizCtl = screen.getByLabelText('Home bar horizontal offset') as HTMLInputElement
    fireEvent.change(horizCtl, { target: { value: '24' } })
    expect(document.documentElement.style.getPropertyValue('--ph-home-topbar-horiz')).toBe('24px')
    const wCtl = screen.getByLabelText('Home bar total width') as HTMLInputElement
    fireEvent.change(wCtl, { target: { value: '300' } })
    expect(document.documentElement.style.getPropertyValue('--ph-home-topbar-width')).toBe('300px')

    // verify page content offsets
    const pVert = screen.getByLabelText('Page vertical offset') as HTMLInputElement
    fireEvent.change(pVert, { target: { value: '12.5' } })
    expect(document.documentElement.style.getPropertyValue('--ph-dashboard-content-offset')).toBe('12.5px')
    const pHoriz = screen.getByLabelText('Page horizontal offset') as HTMLInputElement
    fireEvent.change(pHoriz, { target: { value: '20.25' } })
    expect(document.documentElement.style.getPropertyValue('--ph-dashboard-content-horiz')).toBe('20.25px')
    const pwCtl = screen.getByLabelText('Page content width') as HTMLInputElement
    fireEvent.change(pwCtl, { target: { value: '800' } })
    expect(document.documentElement.style.getPropertyValue('--ph-dashboard-content-width')).toBe('800px')
  })

  it('renders sidebar navigation links', () => {
    // open adjuster panel and verify inputs appear
    const adjBtn = screen.getByLabelText('Adjust topbar edges')
    fireEvent.click(adjBtn)
    expect(screen.getByLabelText('Vertical offset')).toBeInTheDocument()
    expect(screen.getByLabelText('Primary bar left inset')).toBeInTheDocument()
    expect(screen.getByLabelText('Primary bar right inset')).toBeInTheDocument()    // switch to clients page for this scenario
    mockPath = '/practice-hub/clients'
    // Render the layout so sidebar is present
    // simulate stored left margin to verify it's applied on messages page
    document.documentElement.style.setProperty('--ph-page-left-margin', '105.2px')
    render(<PracticeHubLayout>{<PracticeHubPage />}</PracticeHubLayout>)
    // secondary header has updated label as well
    expect(screen.getByText('Clients')).toBeInTheDocument()

    // Navigation items appear in sidebar
    const clientsLinks = screen.getAllByText((content, element) => {
      return element?.textContent?.trim() === 'Clients'
    })
    expect(clientsLinks.length).toBeGreaterThan(0)
    expect(screen.getByText('Home')).toBeInTheDocument()
    // Client Portal group and its sub-items
    expect(screen.getByText('Client Portal')).toBeInTheDocument()
    expect(screen.getByText('Requests')).toBeInTheDocument()
    expect(screen.getByText('Workpapers')).toBeInTheDocument()
    expect(screen.getByText('Team')).toBeInTheDocument()
    expect(screen.getByText('Reports')).toBeInTheDocument()
    expect(screen.getByText('Settings')).toBeInTheDocument()
    // and a top-level Tasks link
    expect(screen.getByText('Tasks')).toBeInTheDocument()

    // Sidebar should no longer use w-fit (fixed width now)
    const aside = document.querySelector('aside.sidebar-floating')
    expect(aside).not.toHaveClass('w-fit')
  })

  it('client tasks page shows calendar/projects/all tasks tabs', () => {
    // Import lazily inside test to avoid bundling issues
    const { default: TasksPage } = require('@/app/practice-hub/tasks/page')
    render(<TasksPage />)

    expect(screen.getByRole('tab', { name: 'Calendar' })).toBeInTheDocument()
    expect(screen.getByRole('tab', { name: 'Projects' })).toBeInTheDocument()
    expect(screen.getByRole('tab', { name: 'All tasks' })).toBeInTheDocument()
    // header should still show title
    expect(screen.getByText('Tasks')).toBeInTheDocument()
  })






  it('renders client list section', () => {
    render(<PracticeHubPage />)
    
    expect(screen.getByText('Your client portfolio at a glance')).toBeInTheDocument()
    // Client names appear both in table and work queue, so use getAllByText
    const riveraEntries = screen.getAllByText('Rivera Enterprises')
    expect(riveraEntries.length).toBeGreaterThan(0)
    const techStartEntries = screen.getAllByText('TechStart Corp')
    expect(techStartEntries.length).toBeGreaterThan(0)
    const northwindEntries = screen.getAllByText('Northwind Labs')
    expect(northwindEntries.length).toBeGreaterThan(0)
  })

  it('renders clients hub tabs', () => {
    mockPath = '/practice-hub/clients'
    render(<PracticeHubPage />)

    expect(screen.getByText('Clients Overview')).toBeInTheDocument()
    expect(screen.getByText('Active Clients')).toBeInTheDocument()
    expect(screen.getByText('Invitations')).toBeInTheDocument()
    expect(screen.getByText('Archived')).toBeInTheDocument()

    // Tablist exists in its own rounded container and should be outside the page container
    const tabList = screen.getByRole('tablist', { name: 'Clients Tabs' })
    expect(tabList).toBeInTheDocument()
    expect(tabList.parentElement).toHaveClass('rounded-2xl')

    // Ensure the tablist is not inside the main Clients page container and the page has a single container
    const pageContainer = screen.getByTestId('clients-page-container')
    expect(pageContainer).toBeInTheDocument()
    // The page container wrapper should have a tighter top margin now
    expect(pageContainer.parentElement).toHaveClass('mt-1')
    // Content wrapper should be right-aligned (ml-auto) to align right edge with topbar
    const contentWrapper = pageContainer.parentElement?.parentElement
    expect(contentWrapper).toHaveClass('ml-auto')
    const clientsHeading = screen.getByText('Clients')
    const pageBar = clientsHeading.closest('[data-testid="clients-page-container"]')
    expect(pageBar).toBeInTheDocument()
    expect(tabList.closest('[data-testid="clients-page-container"]')).toBeNull()
  })

  it('has topbar controls (Invite, New Work, Help, Notifications, Profile) in layout', () => {
    // Set overview offset before render and navigate to practice-hub so PageLayoutDefaults applies it
    localStorage.setItem('ph-clients-overview-offset', '24')
    window.history.pushState({}, '', '/practice-hub')
    render(<PracticeHubLayout>{<PracticeHubPage />}</PracticeHubLayout>)

    const inviteButtons = screen.getAllByText(/\+ Invite Client|\+ Invite/i)
    expect(inviteButtons.length).toBeGreaterThan(0)

    expect(screen.getByText('New Work')).toBeInTheDocument()
    // Help button
    expect(screen.getByLabelText('Help')).toBeInTheDocument()
    // Notifications button
    expect(screen.getByLabelText('Notifications')).toBeInTheDocument()
    // Profile marker (topbar)
    expect(screen.getByLabelText('Profile')).toBeInTheDocument()

    // Layout control should be removed from the topbar and page
    expect(screen.queryByLabelText('Layout')).toBeNull()
    expect(screen.queryByLabelText('Adjust layout')).toBeNull()
    // Secondary bar position adjuster should be present
    const secAdjust = screen.getByLabelText('Adjust secondary bar position')
    expect(secAdjust).toBeInTheDocument()

    // Click to open and ensure the new labels are present
    fireEvent.click(secAdjust)
    expect(screen.getByText('Layout Positions')).toBeInTheDocument()
    expect(screen.getByText('Secondary Bar Offset:')).toBeInTheDocument()
    expect(screen.getByText('Sidebar Offset:')).toBeInTheDocument()
    expect(screen.getByText('Clients Page Offset:')).toBeInTheDocument()
    expect(screen.getByText('Clients Overview Offset:')).toBeInTheDocument()
    expect(screen.getByText('Clients Tab Bar Offset:')).toBeInTheDocument()

    // Overview offset should be applied only to the clients container when set
    const container = screen.getByTestId('clients-page-container')
    expect(document.documentElement.style.getPropertyValue('--ph-clients-overview-offset')).toBe('24px')
    expect(container.style.marginTop).toBe('var(--ph-clients-overview-offset, 0px)')

    // Tabbar should be smaller (reduced padding + text size)
    const tabButton = screen.getByRole('tab', { name: /Clients Overview/i })
    expect(tabButton.className).toMatch(/text-xs/)
    expect(tabButton.className).toMatch(/px-2/)
    expect(tabButton.className).toMatch(/py-0.5/)

    // Tab container wrapper should have reduced padding
    const tabWrapper = tabButton.closest('div.rounded-2xl')
    expect(tabWrapper).toBeTruthy()
    if (tabWrapper) {
      expect(tabWrapper.className).toMatch(/p-2/)
    }
  })
})
