  <script>
    const choiceScreen = document.querySelector('.space-y-6').parentElement;
    const codeScreen = document.getElementById('codeScreen');
    const emailBtn = document.getElementById('emailBtn');
    const phoneBtn = document.getElementById('phoneBtn');
    const backToChoice = document.getElementById('backToChoice');
    const targetDisplay = document.getElementById('targetDisplay');
    const codeForm = document.getElementById('codeForm');
    const codeInputs = codeForm.querySelectorAll('input');

    // Simulate user data
    const userEmail = "user@example.com";
    const userPhone = "+1 (555) 123-4567";

    emailBtn.addEventListener('click', () => {
      targetDisplay.textContent = userEmail;
      document.getElementById('codeInfo').innerHTML = `We sent a 6-digit code to <span class="font-semibold">${userEmail}</span>`;
      choiceScreen.classList.add('hidden');
      codeScreen.classList.remove('hidden');
      codeInputs[0].focus();
    });

    phoneBtn.addEventListener('click', () => {
      targetDisplay.textContent = userPhone;
      document.getElementById('codeInfo').innerHTML = `We sent a 6-digit code to <span class="font-semibold">${userPhone}</span>`;
      choiceScreen.classList.add('hidden');
      codeScreen.classList.remove('hidden');
      codeInputs[0].focus();
    });

    backToChoice.addEventListener('click', () => {
      codeScreen.classList.add('hidden');
      choiceScreen.classList.remove('hidden');
      codeInputs.forEach(i => i.value = '');
    });

    // Code input UX
    codeInputs.forEach((input, i) => {
      input.addEventListener('input', (e) => {
        e.target.value = e.target.value.replace(/\D/g, '');
        if (e.target.value && i < 5) codeInputs[i + 1].focus();
      });
      input.addEventListener('keydown', (e) => {
        if (e.key === 'Backspace' && !e.target.value && i > 0) {
          codeInputs[i - 1].focus();
        }
      });
    });

    codeForm.addEventListener('submit', (e) => {
      e.preventDefault();
      const code = Array.from(codeInputs).map(i => i.value).join('');
      if (code.length === 6) {
        alert('Verification successful! Code: ' + code);
        // In real app: proceed to dashboard
      } else {
        alert('Please enter the full 6-digit code');
      }
    });

    document.getElementById('resendCode').addEventListener('click', () => {
      alert('Code resent!');
    });
  </script>
